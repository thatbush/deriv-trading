# AI Bot (`/bots/ai`) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the `/bots/ai` coming-soon stub with a native React page that uploads a chart screenshot, gets a Gemini Rise/Fall prediction (via a server proxy), and places a Deriv trade using the shell's logged-in account.

**Architecture:** Native page rendered as shell `children` (not an iframe). It consumes the shell's `useAuth` via a new `AuthContext`, calls a new server route `/api/bots/ai/analyze` to keep the Gemini key off the client, and runs the Deriv `authorize→proposal→buy` sequence through a focused, unit-tested `lib/bots/ai-trade.ts` module.

**Tech Stack:** Next.js 16 (App Router), React 19, TypeScript, Tailwind v4, Deriv WebSocket API, Gemini 2.5 Flash, Vitest (added in Task 1).

**Spec:** `docs/superpowers/specs/2026-06-12-ai-bot-design.md`

```ts
import type { AiAnalysis, TradeDirection } from './ai-types';

/** Normalize a free-text timeframe (e.g. "5 minutes") to a compact code ("5m"). */
function normalizeTimeframe(raw: string): string {
  const t = raw.trim().toLowerCase();
  if (t.includes('minute') || t.includes('min')) {
    const m = parseInt(t.match(/\d+/)?.[0] ?? '5', 10) || 5;
    return `${m}m`;
  }
  if (t.includes('hour') || t.includes('hr')) {
    const h = parseInt(t.match(/\d+/)?.[0] ?? '1', 10) || 1;
    return `${h}h`;
  }
  if (t.includes('day')) {
    const d = parseInt(t.match(/\d+/)?.[0] ?? '1', 10) || 1;
    return `${d}d`;
  }
  return '5m';
}

/** Parse Gemini's analysis text into structured fields, with safe fallbacks. */
export function parseAnalysis(text: string): AiAnalysis {
  const market = text.match(/market:?\s*([^,\n]+)/i)?.[1]?.trim() ?? 'Unknown Market';
  const tfMatch = text.match(/timeframe:?\s*([^,\n]+)/i)?.[1];
  const timeframe = tfMatch ? normalizeTimeframe(tfMatch) : '5m';
  const risePercent = parseInt(text.match(/rise:?\s*(\d+)%/i)?.[1] ?? '50', 10);
  const fallPercent = parseInt(text.match(/fall:?\s*(\d+)%/i)?.[1] ?? '50', 10);
  return { market, timeframe, risePercent, fallPercent, text };
}

/** True when the chosen direction contradicts the AI's favored direction. */
export function isAgainstPrediction(
  direction: TradeDirection,
  analysis: Pick<AiAnalysis, 'risePercent' | 'fallPercent'>,
): boolean {
  return (
    (direction === 'RISE' && analysis.fallPercent > analysis.risePercent) ||
    (direction === 'FALL' && analysis.risePercent > analysis.fallPercent)
  );
}
```

```ts
/** Map AI-detected market text to a Deriv trading symbol, or null if unknown. */
export function marketTextToSymbol(marketText: string): string | null {
  const t = marketText.toLowerCase();

  if (t.includes('volatility') || t.includes('vol')) {
    const is1s = t.includes('1s') || t.includes('1hz');
    const sizes = ['10', '25', '50', '75', '100'] as const;
    const extra = ['150', '200', '300'] as const;
    for (const n of sizes) {
      if (t.includes(n)) return is1s ? `1HZ${n}V` : `R_${n}`;
    }
    if (!is1s) {
      for (const n of extra) {
        if (t.includes(n)) return `R_${n}`;
      }
    }
    return null;
  }

  if (t.includes('eur') && t.includes('usd')) return 'frxEURUSD';
  if (t.includes('gbp') && t.includes('usd')) return 'frxGBPUSD';
  if (t.includes('aud') && t.includes('usd')) return 'frxAUDUSD';
  if (t.includes('usd') && t.includes('jpy')) return 'frxUSDJPY';

  if (t.includes('btc') || t.includes('bitcoin')) return 'cryBTCUSD';
  if (t.includes('eth') || t.includes('ethereum')) return 'cryETHUSD';
  if (t.includes('ltc') || t.includes('litecoin')) return 'cryLTCUSD';

  return null;
}

/** Convert a compact timeframe code ("5m"/"2h"/"1d") to whole minutes. */
export function durationToMinutes(timeframe: string): number {
  const value = parseInt(timeframe, 10);
  const unit = timeframe.slice(-1).toLowerCase();
  if (Number.isNaN(value)) return 5;
  switch (unit) {
    case 'm': return value;
    case 'h': return value * 60;
    case 'd': return value * 24 * 60;
    default: return 5;
  }
}
```

NOTE on ordering: the `'100'` check must come before `'10'` would falsely match —
the `sizes` array is ordered `10,25,50,75,100`, and `'Volatility 100'` contains
`'10'`. To avoid `'100'` matching `'10'` first, the test `t.includes('10')` is true
for both. Fix by checking longest-first: reorder `sizes` to `['100','75','50','25','10']`
and `extra` to `['300','200','150']` in the implementation so longer numbers win.

```ts
import type { TradeRequest, TradeResult } from './ai-types';

const APP_ID = '65715';

/**
 * Run authorize → proposal → buy over a fresh OTP WebSocket URL.
 * The URL is single-use (minted by getFreshWsUrl), so callers must pass a fresh one.
 * Resolves with the executed trade's ids; rejects with the Deriv API error message.
 */
export function placeTrade(wsUrl: string, req: TradeRequest): Promise<TradeResult> {
  return new Promise((resolve, reject) => {
    const ws = new WebSocket(wsUrl);
    let settled = false;

    const finish = (fn: () => void) => {
      if (settled) return;
      settled = true;
      fn();
      ws.close();
    };

    ws.onerror = () => finish(() => reject(new Error('WebSocket error occurred')));

    ws.onmessage = (msg) => {
      const res = JSON.parse(msg.data as string);
      if (res.error) {
        finish(() => reject(new Error(res.error.message)));
        return;
      }
      if (res.authorize) {
        ws.send(JSON.stringify({
          proposal: 1,
          amount: req.amount,
          basis: 'stake',
          contract_type: req.direction === 'RISE' ? 'CALL' : 'PUT',
          currency: req.currency,
          symbol: req.symbol,
          duration: req.durationMinutes,
          duration_unit: 'm',
        }));
        return;
      }
      if (res.proposal) {
        ws.send(JSON.stringify({ buy: res.proposal.id, price: res.proposal.ask_price }));
        return;
      }
      if (res.buy) {
        finish(() => resolve({
          transactionId: res.buy.transaction_id,
          longcode: res.buy.longcode,
        }));
      }
    };

    ws.onopen = () => {
      // The OTP URL already encodes auth, but the old app re-authorizes explicitly;
      // the OTP-minted socket authorizes on connect, so go straight to proposal by
      // sending an authorize ping that the OTP socket answers with `authorize`.
      ws.send(JSON.stringify({ authorize: '1' }));
    };
  });
}

export { APP_ID };
```

NOTE: The OTP WebSocket URL minted by `getFreshWsUrl()` authenticates the socket on
connect. During manual verification (Task 9), confirm the `authorize` echo arrives;
if the OTP socket does not respond to `{authorize:'1'}`, switch the `onopen` handler
to send the `proposal` directly (remove the authorize ping) — this is the one point
that depends on Deriv's OTP-socket behavior and must be confirmed live.

```ts
import { NextResponse } from 'next/server';

const GEMINI_URL =
  'https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent';

const PROMPT =
  'Analyze this market chart and predict if it shows an upward or downward trend. ' +
  'Identify the market type (e.g. Volatility 100, EUR/USD, etc) and timeframe if visible. ' +
  'Provide percentage confidence for both rise and fall scenarios. Give a detailed, ' +
  'humanly-written analysis. Start with the format: Market: [market type], ' +
  'Timeframe: [timeframe], Rise: X%, Fall: Y%. Then provide a comprehensive analysis paragraph.';

const sleep = (ms: number) => new Promise((r) => setTimeout(r, ms));

export async function POST(request: Request) {
  const key = process.env.GEMINI_API_KEY;
  if (!key) {
    return NextResponse.json({ error: 'Gemini key not configured' }, { status: 500 });
  }

  let body: { image?: string };
  try {
    body = await request.json();
  } catch {
    return NextResponse.json({ error: 'Invalid JSON body' }, { status: 400 });
  }

  const base64 = body.image?.split(',')[1];
  if (!base64) {
    return NextResponse.json({ error: 'Missing image' }, { status: 400 });
  }

  const payload = {
    contents: [{
      parts: [
        { text: PROMPT },
        { inline_data: { mime_type: 'image/jpeg', data: base64 } },
      ],
    }],
    generationConfig: { temperature: 0.7, topK: 32, topP: 1, maxOutputTokens: 20000 },
  };

  for (let attempt = 1; attempt <= 3; attempt++) {
    try {
      const res = await fetch(GEMINI_URL, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'x-goog-api-key': key },
        body: JSON.stringify(payload),
      });

      if (res.status === 503) {
        if (attempt === 3) {
          return NextResponse.json({ error: 'AI model overloaded, try again later' }, { status: 503 });
        }
        await sleep(Math.min(1000 * 2 ** attempt, 10000));
        continue;
      }
      if (!res.ok) {
        return NextResponse.json({ error: `Gemini error ${res.status}` }, { status: 502 });
      }

      const data = await res.json();
      const text = data?.candidates?.[0]?.content?.parts?.[0]?.text;
      if (typeof text !== 'string') {
        return NextResponse.json({ error: 'Invalid response from AI' }, { status: 502 });
      }
      return NextResponse.json({ text });
    } catch {
      if (attempt === 3) {
        return NextResponse.json({ error: 'Failed to reach AI service' }, { status: 502 });
      }
    }
  }
  return NextResponse.json({ error: 'AI request failed' }, { status: 502 });
}
```

```tsx
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ image }),
      });
      const data = await res.json();
      if (!res.ok) {
        showToast('error', data.error ?? 'Analysis failed');
        return;
      }
      setAnalysis(parseAnalysis(data.text));
    } catch {
      showToast('error', 'Could not reach analysis service');
    } finally {
      setAnalyzing(false);
    }
  };

  const runTrade = async (direction: TradeDirection) => {
    if (!analysis || !account) return;
    const symbol = marketTextToSymbol(analysis.market);
    if (!symbol) {
      showToast('error', `Market "${analysis.market}" not recognized`);
      return;
    }
    if (!(amount >= 1)) {
      showToast('error', 'Enter a valid amount');
      return;
    }
    setPending(direction);
    try {
      const wsUrl = await auth.getFreshWsUrl();
      if (!wsUrl) {
        showToast('error', 'Not authenticated');
        return;
      }
      const result = await placeTrade(wsUrl, {
        direction,
        symbol,
        amount,
        currency: account.currency,
        durationMinutes: durationToMinutes(analysis.timeframe),
      });
      showToast('success', `Trade placed (${direction}) — #${result.transactionId}`);
    } catch (err) {
      showToast('error', err instanceof Error ? err.message : 'Trade failed');
    } finally {
      setPending(null);
    }
  };

  const onTradeClick = (direction: TradeDirection) => {
    if (!analysis) return;
    if (isAgainstPrediction(direction, analysis)) {
      setWarning(direction);
      return;
    }
    runTrade(direction);
  };

  return (
    <div className="flex flex-col min-h-full bg-[var(--background)] text-[var(--foreground)] pb-14">
      <section className="flex flex-col px-6 pt-12 pb-6 max-w-2xl mx-auto w-full">
        <div className="flex items-center gap-3 mb-2">
          <div className="w-10 h-10 rounded-xl bg-violet-100 dark:bg-violet-950 flex items-center justify-center flex-shrink-0">
            <span className="text-violet-600 dark:text-violet-400 text-xl font-bold">✦</span>
          </div>
          <h1 className="text-2xl sm:text-3xl font-semibold tracking-tight">AI Market Analysis</h1>
        </div>
        <p className="text-sm text-zinc-500">
          Upload a chart screenshot. The AI predicts Rise/Fall — then trade on your active account.
        </p>
      </section>

      <section className="px-6 max-w-2xl mx-auto w-full flex flex-col gap-4">
        {/* Account strip */}
        <div className="rounded-xl border border-zinc-200 dark:border-zinc-800 p-3 text-sm">
          {authed && account ? (
            <span>
              {account.account_id} · <span className="font-medium">{account.balance} {account.currency}</span>
            </span>
          ) : (
            <span className="text-zinc-500">Log in (top-right) to place trades.</span>
          )}
        </div>

        {/* Upload */}
        <label className="block cursor-pointer rounded-2xl border-2 border-dashed border-violet-400 h-56 flex items-center justify-center overflow-hidden bg-zinc-50 dark:bg-zinc-900/40">
          {image ? (
            // eslint-disable-next-line @next/next/no-img-element
            <img src={image} alt="Uploaded chart" className="max-w-full max-h-full object-contain" />
          ) : (
            <span className="text-sm text-zinc-500">Tap to upload market chart screenshot</span>
          )}
          <input type="file" accept="image/*" className="hidden" onChange={onFile} />
        </label>

        <button
          onClick={analyze}
          disabled={!image || analyzing}
          className="rounded-xl bg-violet-600 text-white py-3 font-semibold disabled:opacity-50"
        >
          {analyzing ? 'Analyzing…' : 'Analyze Market'}
        </button>

        {/* Prediction */}
        {analysis && (
          <div className="rounded-2xl border border-violet-200 dark:border-violet-900 p-4 flex flex-col gap-3">
            <div className="flex justify-between text-sm">
              <span className="text-violet-600 dark:text-violet-400">{analysis.market}</span>
              <span className="text-zinc-500">{analysis.timeframe}</span>
            </div>
            <div className="flex gap-3 text-sm">
              <span className="text-emerald-500">Rise {analysis.risePercent}%</span>
              <span className="text-red-500">Fall {analysis.fallPercent}%</span>
            </div>
            <p className="text-sm text-zinc-500 whitespace-pre-line">{analysis.text}</p>

            <div className="flex items-center gap-2">
              <label className="text-sm text-zinc-500">Amount</label>
              <input
                type="number" min={1} value={amount}
                onChange={(e) => setAmount(Math.max(1, Number(e.target.value)))}
                className="flex-1 rounded border border-zinc-300 dark:border-zinc-700 bg-transparent px-3 py-2"
              />
              <span className="text-sm text-zinc-500">{account?.currency ?? ''}</span>
            </div>

            <div className="flex gap-3">
              <button
                onClick={() => onTradeClick('RISE')}
                disabled={!authed || pending !== null}
                className="flex-1 rounded-xl border border-emerald-500 text-emerald-600 dark:text-emerald-400 py-3 font-semibold disabled:opacity-50"
              >
                {pending === 'RISE' ? 'Processing…' : `Rise (${analysis.timeframe})`}
              </button>
              <button
                onClick={() => onTradeClick('FALL')}
                disabled={!authed || pending !== null}
                className="flex-1 rounded-xl border border-red-500 text-red-600 dark:text-red-400 py-3 font-semibold disabled:opacity-50"
              >
                {pending === 'FALL' ? 'Processing…' : `Fall (${analysis.timeframe})`}
              </button>
            </div>
          </div>
        )}
      </section>

      {/* Against-prediction warning */}
      {warning && analysis && (
        <div className="fixed inset-0 z-[60] flex items-center justify-center bg-black/60 p-6">
          <div className="rounded-2xl bg-[var(--background)] border border-zinc-200 dark:border-zinc-800 p-6 max-w-sm text-center">
            <p className="text-amber-500 text-3xl mb-2">⚠</p>
            <p className="font-semibold mb-1">Trade against prediction?</p>
            <p className="text-sm text-zinc-500 mb-4">
              AI favors {warning === 'RISE' ? `FALL (${analysis.fallPercent}%)` : `RISE (${analysis.risePercent}%)`} — you chose {warning}.
            </p>
            <div className="flex gap-3">
              <button onClick={() => setWarning(null)} className="flex-1 rounded-lg border border-zinc-300 dark:border-zinc-700 py-2">Cancel</button>
              <button
                onClick={() => { const d = warning; setWarning(null); runTrade(d); }}
                className="flex-1 rounded-lg bg-amber-500 text-white py-2"
              >
                Proceed
              </button>
            </div>
          </div>
        </div>
      )}

      {/* Toast */}
      {toast && (
        <div className={`fixed bottom-6 left-1/2 -translate-x-1/2 z-[70] rounded-lg px-4 py-3 text-sm text-white text-center max-w-[90%] ${toast.kind === 'success' ? 'bg-emerald-600' : 'bg-red-600'}`}>
          {toast.msg}
        </div>
      )}
    </div>
  );
}
```
